\begin{DoxyCode}{0}
\DoxyCodeLine{var\ expand\ =\ require('expand-\/template')()}
\DoxyCodeLine{var\ template\ =\ '\{foo\}/\{foo\}/\{bar\}/\{bar\}'}
\DoxyCodeLine{console.log(expand(template,\ \{}
\DoxyCodeLine{\ \ foo:\ 'BAR',}
\DoxyCodeLine{\ \ bar:\ 'FOO'}
\DoxyCodeLine{\}))}
